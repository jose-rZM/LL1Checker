no terminal axioma;
no terminal listaSent;
no terminal sent;
no terminal whileSent;
no terminal doSent;
no terminal ifSent;
no terminal asigSent;
no terminal elseSent;
no terminal cond;
no terminal var;
terminal WHILE while;
terminal IF "if";
terminal ELSE "else";
terminal AP "(";
terminal CP ")";
terminal DO "do";
terminal PYC ";";
terminal IDENT [a-zA-Z_][a-zA-Z_0-9]*;
terminal ALL "{";
terminal CLL "}";
terminal NUM [1-9][0-9]*;
terminal IGUAL "=";
terminal MENOR "<";
set EOL char "!";
start with axioma;
;
axioma -> listaSent EOL;
listaSent -> sent listaSent;
listaSent ->;
sent -> whileSent;
sent -> doSent;
sent -> ifSent;
sent -> asigSent;
whileSent -> WHILE AP cond CP sent;
doSent -> DO sent WHILE AP cond CP PYC;
ifSent -> IF AP cond CP sent elseSent;
asigSent -> IDENT IGUAL var PYC;
sent -> ALL listaSent CLL;
cond -> var MENOR var;
elseSent-> ELSE sent;
var -> IDENT;
var -> NUM;
;
